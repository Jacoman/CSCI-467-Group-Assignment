USE z1843669;

CREATE TABLE SalesAssociate(UserID INTEGER AUTO_INCREMENT PRIMARY KEY, Name CHAR(255) NOT NULL, Email CHAR(191) UNIQUE NOT NULL, Password CHAR(191) NOT NULL);

CREATE TABLE Customer(CustomerID INTEGER AUTO_INCREMENT PRIMARY KEY, Name CHAR(255) NOT NULL, Email CHAR(191) UNIQUE NOT NULL);

CREATE TABLE Quote(QuoteID INTEGER AUTO_INCREMENT PRIMARY KEY, CustomerID INTEGER, FOREIGN KEY (CustomerID) REFERENCES Customer (CustomerID), UserID INTEGER, FOREIGN KEY (UserID) REFERENCES SalesAssociate (UserID), QuoteName CHAR(255) NOT NULL, Note CHAR(255), Status CHAR(10) NOT NULL DEFAULT 'Ordered', DateTime DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP, Commission CHAR(255));

CREATE TABLE LineItem(LineID INTEGER AUTO_INCREMENT PRIMARY KEY, QuoteID INTEGER, Description CHAR(255) NOT NULL, Price DOUBLE NOT NULL);
